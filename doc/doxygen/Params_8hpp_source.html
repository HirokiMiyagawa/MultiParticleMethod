488  void preCalc() {
489  m_Lref_x = Length_x / Lref;
490  m_Lref_y = Length_y / Lref;
491  m_h0 = h0 / Lref;
492  m_hc0 = m_h0 * 2;
493  m_Pc = Pc / Pref;
494  m_fx = m_kx * m_Lref_x;
495  m_fy = m_ky * m_Lref_x;
496  m_fz = m_kz * m_Lref_x;
497  // XXX: Why add Pref?
498  // m_k = sqrt(m_kx * m_kx + m_ky * m_ky + m_kz * m_kz) + Pref;
499  m_k = sqrt(m_kx * m_kx + m_ky * m_ky + m_kz * m_kz);
500 
501  // C_EI = m_E / (m_rho * Vref * Vref * m_k);
502  C_EI = m_E / (m_rho * Vref * Vref);
503  if (SimplePressure) {
504  C_AE = (Pc * Lref) / (m_E * h0);
505  } else {
506  C_AE = (Pref * Lref) / (m_E * h0);
507  }
508  if (SimpleTensile) {
509  m_Cv = Cv;
510  // m_Cv = Cv * m_k;
511  } else {
512  m_Cv = Cv;
513  }
514 
515  m_preCalc1 = 1 / (1 - (m_nu * m_nu));
516  m_preCalc2 = 1 / (4 * (1 + m_nu));
517  m_preCalc3 = m_nu / (1 - m_nu);
518 
519  // m_preCalc4 = h0 * h0 * m_preCalc1 / ( 12 * Lref * Lref );
520  m_preCalc4 = m_preCalc1 / 12;
521  pre_eta_n = (1 + reflectance * specularity);
522  pre_eta_t = (1 - reflectance * specularity);

528  // rotation ineria system
