101  void readfromstream(std::istream& is) {
102  std::string line;
103  while (getline(is, line)) {
104  if (*(line.end() - 1) == 0x0d) line.pop_back();
105  if (line.length() > 0 && line[0] == '#') {
106  continue;
107  }
108  size_t index = line.find("=");
109  if (std::string::npos != index) {
110  std::string key = line.substr(0, index);
111  std::string value = line.substr(index + 1, line.length());
112  params.insert(ptype::value_type(key, value));
113  }
114  }
115  }
