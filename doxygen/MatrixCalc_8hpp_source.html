418 void MatrixCalc::SortRows(vector<double>& arr, vector<vector<double>>& vec, SortMode const& sort_mode) {
419  int n = arr.size();
420  for (int i = 0; i < n - 1; i++) {
421  int i_max = i;
422  for (int j = i + 1; j < n; j++) {
423  // find the "maximum" element in the array starting at position i+1
424  switch (sort_mode) {
425  case SORT_DECREASING_ARRAY:
426  if (arr[j] > arr[i_max])
427  i_max = j;
428  break;
429  case SORT_INCREASING_ARRAY:
430  if (arr[j] < arr[i_max])
431  i_max = j;
432  break;
433  case SORT_DECREASING_ABS_ARRAY:
434  if (std::abs(arr[j]) > std::abs(arr[i_max]))
435  i_max = j;
436  break;
437  case SORT_INCREASING_ABS_ARRAY:
438  if (std::abs(arr[j]) < std::abs(arr[i_max]))
439  i_max = j;
440  break;
441  default:
442  break;
443  }
444  }
445  std::swap(arr[i], arr[i_max]); // sort "arr"
446  for (int k = 0; k < n; k++)
447  std::swap(vec[i][k], vec[i_max][k]); // sort "vec"
448  }
449 }
